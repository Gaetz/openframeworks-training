var num:Number = 0;
for (i=0; i<10; i++) {
	for (j=0; j<10; j++) {
		var nm:MovieClip = attachMovie("circle", "cir"+num, num);
		nm._xscale = 171;
		nm._yscale = 171;
		nm._x = i * 62.5;
		nm._y = j * 62.5;
		nm.sx = nm._x;
		nm.sy = nm._y;
		nm.ang = 0;
		nm.mynum = num;
		num++;
		nm.onEnterFrame = function() {
			this._x = this.sx + Math.sin(this.ang) * this.mynum
			this._y = this.sy + Math.sin(this.ang) * this.mynum;
			this.ang += 0.1;
		};
	}
}
