var num:Number = 0;
for (i=0; i<10; i++) {
	for (j=0; j<10; j++) {
		var nm:MovieClip = attachMovie("circle", "cir"+num, num);
		nm._xscale = 171;
		nm._yscale = 171;
		nm._x = i * nm._width;
		nm._y = j * nm._height;
		nm.sx = nm._xscale;
		nm.sy = nm._yscale;
		nm.ang = 0;
		nm.mynum = num;
		num++;
		nm.onEnterFrame = function() {
			this._yscale = this._xscale = this.sx + (Math.sin(this.ang) * this.mynum);
			this.ang += this.mynum / 450;
		};
	}
}
